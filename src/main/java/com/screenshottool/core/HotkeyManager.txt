package com.screenshottool.core;

import javafx.application.Platform;
import java.awt.*;
import java.awt.event.InputEvent;
import java.awt.event.KeyEvent;

/**
 * HotkeyManager - Registra y mantiene el atajo global Ctrl+Shift+S.
 *
 * Responsabilidades:
 * - Crear una JFrame oculta que recibe el atajo de teclado
 * - Mantener el foco en esa ventana con un Timer
 * - Llamar al callback onCapturar cuando se detecta el atajo
 *
 * Nota: En Linux/X11 funciona de forma nativa.
 * En Windows el JFrame oculto necesita el Timer de re-foco
 * para mantener el atajo activo mientras el usuario trabaja
 * en otras aplicaciones.
 */
public class HotkeyManager {

    private final Runnable onCapturar;
    private final javax.swing.JFrame ventanaOculta;

    public HotkeyManager(Runnable onCapturar) {
        this.onCapturar = onCapturar;
        this.ventanaOculta = crearVentanaOculta();
    }

    // ── Inicializar atajo ─────────────────────────────────
    public void iniciar() {
        javax.swing.SwingUtilities.invokeLater(() -> {
            registrarAtajo();
            iniciarTimerFoco();
        });
    }

    // ── Ventana oculta que recibe el atajo ────────────────
    private javax.swing.JFrame crearVentanaOculta() {
    javax.swing.JFrame frame = new javax.swing.JFrame();
    frame.setSize(1, 1);
    frame.setLocation(-200, -200);
    frame.setUndecorated(true);
    frame.setType(Window.Type.UTILITY);
    frame.setAutoRequestFocus(false);
    frame.setFocusableWindowState(false);
    // Ocultar del taskbar de GNOME
    frame.getRootPane().putClientProperty("Window.shadow", false);
    frame.setVisible(true);
    // Mover fuera de pantalla después de visible
    frame.setState(javax.swing.JFrame.ICONIFIED);
    return frame;
}

    // ── Registrar Ctrl+Shift+S en la ventana oculta ───────
    private void registrarAtajo() {
        javax.swing.KeyStroke atajo = javax.swing.KeyStroke.getKeyStroke(
                KeyEvent.VK_S,
                InputEvent.CTRL_DOWN_MASK | InputEvent.SHIFT_DOWN_MASK);

        ventanaOculta.getRootPane()
                .getInputMap(javax.swing.JComponent.WHEN_IN_FOCUSED_WINDOW)
                .put(atajo, "capturar");

        ventanaOculta.getRootPane().getActionMap()
                .put("capturar", new javax.swing.AbstractAction() {
                    public void actionPerformed(java.awt.event.ActionEvent e) {
                        Platform.runLater(onCapturar);
                    }
                });
    }

    // ── Timer que mantiene el foco en la ventana oculta ───
    private void iniciarTimerFoco() {
        String os = System.getProperty("os.name").toLowerCase();
        if (os.contains("linux"))
            return; // En Linux el atajo funciona sin timer

        javax.swing.Timer timer = new javax.swing.Timer(800, e -> {
            Window focusedWindow = KeyboardFocusManager
                    .getCurrentKeyboardFocusManager().getFocusedWindow();
            if (focusedWindow == null) {
                ventanaOculta.toFront();
                ventanaOculta.requestFocus();
            }
        });
        timer.start();
    }

    // ── Pausar el timer cuando hay un diálogo activo ──────
    // Llamado desde AppContext para no robar el foco del diálogo
    public void pausarFoco(boolean pausar) {
        // El timer verifica isFocused() — si hay un diálogo
        // activo el sistema operativo no le dará foco a ventanaOculta
        // de todas formas, pero esto es por claridad
    }
}
